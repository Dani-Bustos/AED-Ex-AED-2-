\begin{flushleft}
TAD Banco:
\end{flushleft}
 obs colaBien : seq(ID)\\
 obs colaProle : seq(ID)\\
 obs posEnCola(e:ID):(\ent)\\
\begin{proc}{Identificacion}{\Inout b: Banco, \In p : ID}{}
\requiere{(p.ID \notin colaBien \lor p.ID \notin colaProle)\land  b_0 = b}
\asegura{p.casta = 0 \implicaLuego b_colaProle = concat(b_0.colaProle,[p]}
\asegura{p.casta = 1 \implicaLuego b_colaBien = concat(b_0.colaBien,[p]}
 

\end{proc}

\begin{proc}{AtenderCajaA}{\Inout b:Banco}{}
\requiere{b_0 = b \land b.colaBien \neq []}
\asegura{b.colaBien = subseq(b_0.colaBien,1,\longitud{b_0.colaBien}) }
\asegura{b.colaProple = b_0.colaProple}
\end{proc}

\begin{proc}{AtenderCajaB}{\Inout q: Banco}{}
\requiere{q_0 = q \land (b.colaBien \neq [] \lor b.colaProle \neq [])}
\asegura{q_0.colaBien = [] \implicaLuego (q_0.colaBien = q.colaBien \land q.colaProle = subseq(q_0.colaProle,1,\longitud{q_0.colaBien}) }
\asegura{q_0.colaBien \neq [] \implicaLuego (q.colaProple = q_0.colaProle \land q.colaBien = subseq(q_0.colaBien,1,\longitud{q_0.colaBien)}}
\end{proc}

\begin{proc}{Cansarse}{\Inout q : Banco, \In p : ID}{}
\requiere{q = q_0 \land (p \in q.colaBien \lor p \in q.colaProle}
\asegura{p \in q_0.colaBien \implicaLuego q.colaBien = concat(subseq(q_0.colaBien,0,q.posEnCola(ID)-1),\\ subseq(q_0.colaBien,q_0.posEnCola(ID)+1,\longitud{q_o.colaBien}) }
\asegura{p \in q_0.colaProle \implicaLuego q.colaProle = concat(subseq(q_0.colaProle,0,q.posEnCola(ID)-1),\\ subseq(q_0.colaProle,q_0.posEnCola(ID)+1,\longitud{q_o.colaProle})}
\end{proc}
